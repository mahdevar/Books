\newcommand\EndNoteMark[1]{\addtocounter{EndNoteCounter}{#1}{\color{footnote}\textsuperscript{\hyperlink{\roman{EndNoteSeries}@\roman{EndNoteCounter}}{\Abjad{EndNoteCounter}}}}\addtocounter{EndNoteCounter}{-#1}}

\newcommand{\WriteFootnotes}
{%
	\ifnum\value{EndNoteCounter}>0%
		\parskip=0pt%
		\par%
		\footnotesize%
		\advance\rightskip\tabcolsep%
		\advance\parindent-\tabcolsep%
		\setcounter{EndNoteCounter}{0}%
		\vskip\skip\@mpfootins%
%		\footnoterule%
		\color{footnote}%
		\renewcommand*{\do}[1]{\makebox[\tabcolsep][r]{\stepcounter{EndNoteCounter}\raisebox{\ht\strutbox}{\hypertarget{\roman{EndNoteSeries}@\roman{EndNoteCounter}}{}}\textsuperscript{\Abjad{EndNoteCounter}}}##1\par}%
		\dolistloop{\EndNoteCounterText}%
		\gdef\EndNoteCounterText{}%
		\setcounter{EndNoteCounter}{0}%
		\stepcounter{EndNoteSeries}%
	\fi%
}

\newcommand\StyleEnvironment[1]
{
	\csappto{#1}
	{
		\Kayhan
		\parindent=\@default@parindent
		\parskip=\@default@parskip
		\enable@lines
		\capstart
		\let\word\wordtext
	}
	\cspreto{#1}{\colorlet{environment}{#1}}
	\cspreto{end#1}
	{
		\WriteFootnotes
		\let\word\@original@word
		\disable@lines
	}
}

\newcommand{\FloatEnvironment}[2]
{
	\renewenvironment{#1}
	{
		\captionsetup{type=#1}
		\begin{tcolorbox}[floats, color=environment]
	}
	{
		\end{tcolorbox}
	}
	\renewenvironment{#1*}
	{
		\captionsetup{type=#1}
		\begin{tcolorbox}[floats, float*, width=\textwidth, color=environment]
	}
	{
		\end{tcolorbox}
	}
	\StyleEnvironment{#1}
	\StyleEnvironment{#1*}
}

\newcommand{\InlineEnvironment}[2]
{
	\renewenvironment{#1}
	{
		\captionsetup{type=#1}
		\begin{tcolorbox}[inline, color=environment]
	}
	{
		\end{tcolorbox}
	}
	\StyleEnvironment{#1}
	\StyleEnvironment{#1*}
}

%MAKE NEW ENVIRONMENTS
\newcommand{\NewEnvironment}[4][]
{
	\newfloat{#2}{b}{#2}[chapter]
	\listadd{\Environments}{#2}
	\definecolor{#2}{RGB}{#4}
	\colorlet{#2*}{#2}
	
	\IfSubStr{#1}{Inline}
		{\InlineEnvironment{#2}}
		{\FloatEnvironment{#2}}

	\expandafter\let\csname
	l@#2\endcsname\l@figure

	\Name{#2}{#3}{#2}
	\Name{list#2}{فهرست #3‌ها}{#2}
	\floatname{#2}{\csname#2name\endcsname}

	\cspreto{#2}{\write\@mainaux{\listxadd{\string\UsedEnvironment}{#2}}}
	\cspreto{#2*}{\write\@mainaux{\listxadd{\string\UsedEnvironment}{#2}}}

	\IfSubStr{#1}{No List}
	{}
	{
		\AtBeginDocument
		{
			\ifinlist{#2}{\UsedEnvironment}
			{\appto\@otherlists{\printlistof{#2}}}
			{}
		}
	}
}
